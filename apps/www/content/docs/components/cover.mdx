## Usage

```tsx
import { Cover, CoverFallback, CoverImage } from "@/components/ui/cover"
```

```tsx
<Cover>
  <CoverImage src="/covers/01.png" alt="Sample Cover" />
  <CoverFallback>SC</CoverFallback>
</Cover>
```

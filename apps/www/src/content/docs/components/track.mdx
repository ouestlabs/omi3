## Usage

```tsx
import { AudioTrack, AudioTrackList } from "@/components/ui/audio/track"
```

### AudioTrack

Display a single track with cover art, metadata, and playback controls.

<ComponentPreview name="track-demo" />

```tsx
// Store mode - loads from global queue
<AudioTrack trackId={track.id} />

// Controlled mode - uses provided data
<AudioTrack track={customTrack} onClick={() => handlePlay(customTrack)} />
```

### AudioTrackList

Display a list of tracks with filtering, reordering, and selection.

<ComponentPreview name="track-list-demo" />

```tsx
// Store mode - displays global queue
<AudioTrackList onTrackSelect={(index) => console.log(index)} />

// Controlled mode - displays custom tracks
<AudioTrackList tracks={searchResults} onTrackSelect={handleSelect} />
```

## API Reference

### AudioTrack

| Prop            | Type                      | Default | Description                          |
| --------------- | ------------------------- | ------- | ------------------------------------ |
| trackId         | `string \| number`        | -       | Track ID to load from store          |
| track           | `Track`                   | -       | Track object (controlled mode)       |
| index           | `number`                  | -       | Display index                        |
| onClick         | `() => void`              | -       | Click handler                        |
| onRemove        | `(id: string) => void`    | -       | Remove handler                       |
| showRemove      | `boolean`                 | `false` | Show remove button                   |
| showPlayPause   | `boolean`                 | `true`  | Show play/pause button               |
| showDragHandle  | `boolean`                 | `false` | Show drag handle                     |
| showCover       | `boolean`                 | `true`  | Show cover image                     |
| className       | `string`                  | -       | Additional classes                   |

**Note:** Provide either `trackId` (store mode) or `track` (controlled mode), not both.

### AudioTrackList

| Prop              | Type                      | Default                | Description                          |
| ----------------- | ------------------------- | ---------------------- | ------------------------------------ |
| tracks            | `Track[]`                 | -                      | Custom tracks (controlled mode)      |
| onTrackSelect     | `(index: number) => void` | -                      | Selection handler                    |
| onTrackRemove     | `(id: string) => void`    | -                      | Remove handler                       |
| sortable          | `boolean`                 | `false`                | Enable drag-and-drop                 |
| showCover         | `boolean`                 | `true`                 | Show cover images                    |
| variant           | `"default" \| "grid"`     | `"default"`            | Layout variant                       |
| filterQuery       | `string`                  | -                      | Filter by title/artist               |
| filterFn          | `(track: Track) => boolean` | -                    | Custom filter function               |
| emptyLabel        | `string`                  | `"No tracks found"`    | Empty state label                    |
| emptyDescription  | `string`                  | `"Try adding some tracks"` | Empty state description          |
| className         | `string`                  | -                      | Additional classes                   |

## Modes

### Store Mode
Reads from and syncs with the global audio store. Use for main queue displays.

```tsx
<AudioTrackList sortable onTrackSelect={(i) => console.log(i)} />
```

### Controlled Mode
Uses provided track data without store integration. Use for search results, browsing, or custom lists.

```tsx
<AudioTrackList
  tracks={externalTracks}
  onTrackSelect={(i) => handleAddToQueue(externalTracks[i])}
/>
```

## Examples

### With Filtering

```tsx
<AudioTrackList
  filterQuery="jazz"
  // or
  filterFn={(track) => track.genre === "jazz"}
/>
```

### With Removal

```tsx
<AudioTrackList
  onTrackRemove={(id) => removeFromQueue(id)}
/>
```

### Full Integration

```tsx
<AudioProvider tracks={initialTracks}>
  <AudioPlayer>
    <AudioControlBar>
      <AudioPlay />
      <AudioSeekBar />
    </AudioControlBar>
    
    {/* Main queue */}
    <AudioTrackList sortable variant="default" />
    
    {/* Search results */}
    <AudioTrackList
      tracks={searchResults}
      variant="grid"
      onTrackSelect={handleAddToQueue}
    />
  </AudioPlayer>
</AudioProvider>
```

## Notes

- `AudioProvider` required for store mode
- Cover images sourced from `track.artwork` or `track.images[0]`
- Live tracks display "LIVE" badge and hide duration
- Sortable disabled when filtering active
- Grid layout responsive (1 column mobile, 2 desktop)
